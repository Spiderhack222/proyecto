#include "algoritmos.h"
#include <stdlib.h> // Para malloc y free

void mezclar(int arr[], int izq, int med, int der, Metricas* m) {
    int i, j, k;
    int n1 = med - izq + 1;
    int n2 = der - med;
    int* L = (int*)malloc(n1 * sizeof(int));
    int* R = (int*)malloc(n2 * sizeof(int));
    for (i = 0; i < n1; i++) { L[i] = arr[izq + i]; m->movimientos++; }
    for (j = 0; j < n2; j++) { R[j] = arr[med + 1 + j]; m->movimientos++; }
    i = 0; j = 0; k = izq;
    while (i < n1 && j < n2) {
        m->comparaciones++;
        if (L[i] <= R[j]) { arr[k] = L[i]; m->movimientos++; i++; }
        else { arr[k] = R[j]; m->movimientos++; j++; }
        k++;
    }
    while (i < n1) { arr[k] = L[i]; m->movimientos++; i++; k++; }
    while (j < n2) { arr[k] = R[j]; m->movimientos++; j++; k++; }
    free(L); free(R);
}

void ordenamientoMezcla(int arr[], int izq, int der, Metricas* m) {
    if (izq < der) {
        int med = izq + (der - izq) / 2;
        ordenamientoMezcla(arr, izq, med, m);
        ordenamientoMezcla(arr, med + 1, der, m);
        mezclar(arr, izq, med, der, m);
    }
}
